```cpp
using namespace std;

class LinkedList {
public:
  int value;
  LinkedList *next;

  LinkedList(int value) {
    this->value = value;
    this->next = NULL;
  }
};

LinkedList *reverseLinkedList(LinkedList *head) {
	
	LinkedList *p = head;
	LinkedList *q = NULL;
	LinkedList *r = NULL;
	
	while(p!=NULL)
	{
		r=q;
		q=p;
		p=p->next;
		cout<<q->value<<endl;
		q->next = r;
	}
	
	head = q;
	
  return q;
}
```

## Optimal Time Space Complexity :

O(n) time | O(1) space - where n is the number of nodes in the Linked List
