class Node:
    def __init__(self,value):
        self.value = value
        self.left = None
        self.right = None

class BinaryTree:
    def __init__(self):
        self.root = None

    def insert(self,node,value):

        if node is None:
            newNode = Node(value)
            if self.root is None:
                self.root = newNode
            return newNode
        elif value > node.value:
            node.right = self.insert(node.right,value)
        elif value < node.value:
            node.left = self.insert(node.left,value)

        return node
